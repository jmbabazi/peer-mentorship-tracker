Connect to Google Sheets

This links the tracker to a shared Google Sheet so you and your manager always see the same data. Follow the steps below — it takes about 5 minutes.

1
Create a new Google Sheet
Go to sheets.new to open a blank spreadsheet. Name it "M26 Mentorship Tracker".
2
Open Apps Script
In your Google Sheet, click Extensions → Apps Script. Delete any existing code and paste the code below:
// M26 Mentorship Tracker — Apps Script Backend // Paste this entire block, then click Deploy const SHEET_NAME = 'Mentors'; const HEADERS = ['id','name','group','status','t1','t2','t3','mentees','wsLed','wsVol','comments']; function initSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) sheet = ss.insertSheet(SHEET_NAME); if (sheet.getLastRow() === 0) { sheet.getRange(1,1,1,HEADERS.length).setValues([HEADERS]); sheet.getRange(1,1,1,HEADERS.length).setFontWeight('bold'); } return sheet; } function doGet(e) { const sheet = initSheet(); const last = sheet.getLastRow(); if (last <= 1) return out([]); const vals = sheet.getRange(2,1,last-1,HEADERS.length).getValues(); const rows = vals.map(row => { const obj = {}; HEADERS.forEach((h,i) => { let v = row[i]; if (v === 'true') v = true; else if (v === 'false') v = false; if (['id','mentees','wsLed','wsVol'].includes(h)) v = Number(v)||0; obj[h] = v; }); return obj; }); return out(rows); } function doPost(e) { try { const { mentors, nextId } = JSON.parse(e.postData.contents); const sheet = initSheet(); const last = sheet.getLastRow(); if (last > 1) sheet.deleteRows(2, last - 1); if (mentors && mentors.length > 0) { const rows = mentors.map(m => HEADERS.map(h => m[h] !== undefined ? m[h] : '')); sheet.getRange(2,1,rows.length,HEADERS.length).setValues(rows); } return out({ok: true}); } catch(err) { return out({error: String(err)}); } } function out(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
